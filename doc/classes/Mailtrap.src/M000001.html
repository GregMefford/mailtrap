  def initialize( host, port, once, msgdir )
    @host = host
    @port = port
    @once = once
    @msgdir = msgdir
    
    puts "Mailtrap starting at #{@host}:#{port} and writing to #{@msgdir}"
    service = TCPServer.new( @host, @port )
    accept( service )
  end
